### Initial Solution
```typescript
function romanToInt(s: string): number {
    const nums = {
        "I": 1,
        "V": 5,
        "X": 10,
        "L": 50,
        "C": 100,
        "D": 500,
        "M": 1000
    }
    
    const sub = {
        "V": "I",
        "X": "I",
        "L": "X",
        "C": "X",
        "D": "C",
        "M": "C"
    }
    
    let sum = 0;
    
    for (let i=0; i<s.length; i++) {
        if (s[i+1] && s[i+1] in sub && s[i] == sub[s[i+1]]) {
            sum += nums[s[i+1]] - nums[s[i]];
            i++;
        } else {
            sum += nums[s[i]];
        }
    }
    
    return sum
};
```

```typescript
let symbols = {
    "I": 1,
    "V": 5,
    "X": 10,
    "L": 50,

function romanToInt(s: string): number {
    let sum = 0;
    
    for (let i=0; i<s.length; i++) {
        symbols[s[i]] < symbols[s[i+1]] 
            ? sum -= symbols[s[i]]
            : sum += symbols[s[i]]
    }
    
    return sum;
};
```
